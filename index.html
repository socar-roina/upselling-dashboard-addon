WITH rsv AS ( SELECT reservation_id, SUM(amount) AS net_amount FROM `socar-data.tianjin_replica.charged_info` WHERE DATE(created_at,"Asia/Seoul") >= "2026-06-05" AND charge_type IN ("addon_carseat","addon_stroller") GROUP BY reservation_id ), zero_rsv AS ( SELECT reservation_id FROM rsv WHERE net_amount = 0 ), gap AS ( SELECT c.reservation_id, TIMESTAMP_DIFF(MAX(IF(c.amount<0,c.created_at,NULL)), MIN(IF(c.amount>0,c.created_at,NULL)), MINUTE) AS diff_min FROM `socar-data.tianjin_replica.charged_info` c JOIN zero_rsv z USING(reservation_id) WHERE c.charge_type IN ("addon_carseat","addon_stroller") GROUP BY c.reservation_id ) SELECT ri.state, COUNT(*) AS rsv_cnt, COUNTIF(ri.cancel_at IS NOT NULL) AS has_cancel_at, ROUND(AVG(g.diff_min),1) AS avg_refund_min FROM gap g JOIN `socar-data.dbt.stg_tianjin_replica__reservation_info` ri ON ri.id = g.reservation_id GROUP BY ri.state ORDER BY ri.state
